Private Sub Worksheet_Activate()
    ActualizarListasComboBoxes
End Sub

Private Sub Worksheet_SelectionChange(ByVal Target As Range)
    If Target.CountLarge > 1 Then Exit Sub
    
    OcultarTodosComboBox

    Dim cmb As OLEObject
    Dim listaRango As Range

    Select Case Target.Address
        Case "$F$3" ' cmb1 - RUC
            Set cmb = Me.OLEObjects("cmb1")
            Set listaRango = ThisWorkbook.Sheets("Registro Clientes").Range("nCRUC")
        Case "$F$6" ' cmb2 - Tipo Documento
            Set cmb = Me.OLEObjects("cmb2")
            Set listaRango = ThisWorkbook.Sheets("Ajustes").Range("nTDocumento")
        Case "$F$4" ' cmb3 - CÛdigo Cliente
            Set cmb = Me.OLEObjects("cmb3")
            Set listaRango = ThisWorkbook.Sheets("Registro Clientes").Range("nCÛdigo_Cliente")
        Case Else
            Exit Sub
    End Select

    With cmb
        .Visible = True
        .Top = Target.Top
        .Left = Target.Left
        .Width = Target.Width + 15
        .Height = Target.Height
        .LinkedCell = Target.Address
        .ListFillRange = ""
        .Object.Clear
        CargarLista cmb, listaRango
        .Object.MatchEntry = fmMatchEntryNone
        .Object.DropDown
    End With
End Sub

Private Sub OcultarTodosComboBox()
    Me.OLEObjects("cmb1").Visible = True
    Me.OLEObjects("cmb2").Visible = True
    Me.OLEObjects("cmb3").Visible = True
End Sub

Public Sub ActualizarListasComboBoxes()
    CargarLista Me.OLEObjects("cmb1"), ThisWorkbook.Sheets("Registro Clientes").Range("nCRUC")
    Me.OLEObjects("cmb1").Visible = True

    CargarLista Me.OLEObjects("cmb2"), ThisWorkbook.Sheets("Ajustes").Range("nTDocumento")
    Me.OLEObjects("cmb2").Visible = True

    CargarLista Me.OLEObjects("cmb3"), ThisWorkbook.Sheets("Registro Clientes").Range("nCÛdigo_Cliente")
    Me.OLEObjects("cmb3").Visible = True
End Sub

Private Sub CargarLista(cmb As OLEObject, rng As Range)
    Dim datos As Variant
    cmb.Object.Clear

    If rng.Cells.Count = 1 Then
        If IsEmpty(rng.Value) Then
            cmb.Object.List = Array("")
        Else
            cmb.Object.List = Array(rng.Value)
        End If
    Else
        datos = rng.Value
        On Error Resume Next
        cmb.Object.List = Application.Transpose(datos)
        On Error GoTo 0
    End If
End Sub

    If KeyCode = 9 Then ActiveCell.Offset(0, 1).Activate

    If KeyCode = 13 Then ActiveCell.Offset(1, 0).Activate
End Sub
