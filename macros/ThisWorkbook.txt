VERSION 1.0 CLASS
BEGIN
  MultiUse = -1  'True
END
Attribute VB_Name = "ThisWorkbook"
Attribute VB_GlobalNameSpace = False
Attribute VB_Creatable = False
Attribute VB_PredeclaredId = True
Attribute VB_Exposed = True
Private Sub Workbook_Open()

    ' === Cargar listas para Registro Productos ===
    With Sheets("Registro Productos")
        If Not .OLEObjects("cmb1") Is Nothing Then
            .OLEObjects("cmb1").Object.List = FiltrarLista(Sheets("Registro Clientes").Range("B2:B1000"), "")
        End If
        If Not .OLEObjects("cmb2") Is Nothing Then
            .OLEObjects("cmb2").Object.List = FiltrarLista(Sheets("Ajustes").Range("B2:B100"), "")
        End If
        If Not .OLEObjects("cmb3") Is Nothing Then
            .OLEObjects("cmb3").Object.List = FiltrarLista(Sheets("Registro Clientes").Range("C2:C1000"), "")
        End If
    End With

    ' === Cargar listas para Registro Clientes ===
    Call Modulo_Regis_Cliente.ActualizarListasComboBoxes_Clientes

    ' === Inicializar ComboBoxes en Nota de Ingreso ===
    Call Modulo_Nota_Ingreso.ActualizarListasNotaIngreso

    ' === Inicializar ComboBoxes en Nota de Salida ===
    Call Modulo_Nota_Salida.ActualizarListasNotaSalida

    ' === Inicializar ComboBoxes en Acta de Recepción ===
    Call Modulo_Acta_Recepcion.ActualizarListasActaRecepcion

End Sub
